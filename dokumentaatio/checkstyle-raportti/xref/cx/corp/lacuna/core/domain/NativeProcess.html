3   /**
4    * Represents a native process running on the host computer.
5    *
6    * <p>This model object contains the process id, owner name and description
7    * of the represented process.
8    * @see cx.corp.lacuna.core.NativeProcessEnumerator
9    * @see cx.corp.lacuna.core.NativeProcessCollector
10   */
11  public interface NativeProcess {
12      /**
13       * Value used to indicate that the owner of this process is unknown.
14       */
15      String UNKNOWN_OWNER = "";
16  
17      /**
18       * Value used to indicate that the description of this process is unknown.
19       */
20      String UNKNOWN_DESCRIPTION = "";
21  
22      /**
23       * Gets the process identifier of the process.
24       */
25      int getPid();
26  
27      /**
28       * Sets the process identifier of the process.
29       */
30      void setPid(int pid);
31  
32      /**
33       * Gets the description of the process.
34       *
35       * <p>On Linux platforms, this value
36       * is the <i>command line</i> of the running process, whereas on Windows
37       * platforms this value is the image name of the process's executable.
38       *
39       * <p>If the process cannot be accessed, this value may be null or empty.
40       * @return The description of the process, or {@link #UNKNOWN_DESCRIPTION} if
41       *         the description is unknown.
42       */
43      String getDescription();
44  
45      /**
46       * Sets the description of the process.
47       */
48      void setDescription(String desc);
49  
50      /**
51       * Gets the owner of the process.
52       *
53       * <p>If the process cannot be accessed, this value may be null or empty.
54       * @return The owner of the process, or {@link #UNKNOWN_OWNER} if
55       *         the owner is unknown.
56       */
57      String getOwner();
58  
59      /**
60       * Sets the name of the process's owner.
61       */
62      void setOwner(String owner);
63  }
