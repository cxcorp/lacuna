1   package cx.corp.lacuna.core.windows;

3   import cx.corp.lacuna.core.windows.winapi.Advapi32;
4   import cx.corp.lacuna.core.windows.winapi.CloseHandle;
5   
6   import java.util.Optional;
7   
8   public class WindowsProcessOwnerGetter implements ProcessOwnerGetter {
9   
10      private final ProcessTokenOpener tokenOpener;
11      private final TokenUserFinder tokenUserFinder;
12      private final TokenOwnerNameFinder tokenOwnerFinder;
13  
14      public WindowsProcessOwnerGetter(ProcessTokenOpener tokenOpener,
15                                       TokenUserFinder userFinder,
16                                       TokenOwnerNameFinder nameFinder) {
17          if (tokenOpener == null || userFinder == null || nameFinder == null) {
18              throw new IllegalArgumentException("Args can't be null");
19          }
20          this.tokenOpener = tokenOpener;
21          this.tokenUserFinder = userFinder;
22          this.tokenOwnerFinder = nameFinder;
23      }
24  
25      @Override
26      public Optional<String> get(ProcessHandle processHandle) {
27          if (processHandle == null) {
28              throw new IllegalArgumentException("processHandle cannot be null!");
29          }
30  
31          try (ProcessToken token = tokenOpener.openToken(processHandle)) {
32              return tokenUserFinder.findTokenUser(token).flatMap(tokenOwnerFinder::getUserName);
33          } catch (TokenOpenException ex) {
34              // loggerino
35              return Optional.empty();
36          }
37      }
38  }
