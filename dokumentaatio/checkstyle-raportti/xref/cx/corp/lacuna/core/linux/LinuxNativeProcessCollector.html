3   import cx.corp.lacuna.core.NativeProcessCollector;
4   import cx.corp.lacuna.core.domain.NativeProcess;
5   import cx.corp.lacuna.core.domain.NativeProcessImpl;
6   
7   import java.io.File;
8   import java.io.IOException;
9   import java.nio.charset.StandardCharsets;
10  import java.nio.file.Files;
11  import java.nio.file.Path;
12  import java.nio.file.Paths;
13  import java.util.List;
14  
15  public class LinuxNativeProcessCollector implements NativeProcessCollector {
16  
17      private static final String PROC_CMDLINE_PATH = "cmdline";
18      private final Path procRoot;
19  
20      public LinuxNativeProcessCollector(Path procRoot) {
21          this.procRoot = procRoot;
22      }
23  
24      @Override
25      public NativeProcess collect(int pid) {
26          Path procPath = procRoot.resolve(Integer.toUnsignedString(pid));
27          return readNativeProcess(procPath.toFile());
28      }
29  
30      private NativeProcess readNativeProcess(File procDirectory) {
31          NativeProcess process = new NativeProcessImpl();
32  
33          String directoryName = procDirectory.getName();
34          int pid = Integer.parseInt(directoryName);
35          process.setPid(pid);
36  
37          String processCmdLine = readCmdLine(directoryName);
38          process.setDescription(processCmdLine);
39  
40          String owner = readOwner(procDirectory.toPath());
41          process.setOwner(owner);
42  
43          return process;
44      }
45  
46      private Path constructCmndLinePath(String pid) {
47          Path childPath = Paths.get(pid, PROC_CMDLINE_PATH);
48          return procRoot.resolve(childPath);
49      }
50  
51      private String readCmdLine(String pidString) {
52          // get process command line for description
53          // "/proc/[pid]/cmdline
54          //     This read-only file holds the complete command line for the
55          //     process, unless the process is a zombie.  In the latter case,
56          //     there is nothing in this file: that is, a read on this file
57          //     will return 0 characters.  The command-line arguments appear
58          //     in this file as a set of strings separated by null bytes
59          //     ('\0'), with a further null byte after the last string."
60          Path cmdLinePath = constructCmndLinePath(pidString);
61          try {
62              List<String> lines = Files.readAllLines(cmdLinePath, StandardCharsets.UTF_8);
63              return lines.size() > 0
64                  ? lines.get(0).replace('\0', ' ').trim()
65                  : NativeProcess.UNKNOWN_DESCRIPTION;
66          } catch (IOException e) {
67              return NativeProcess.UNKNOWN_DESCRIPTION;
68          }
69      }
70  
71      private String readOwner(Path procDirectory) {
72          try {
73              return Files.getOwner(procDirectory).getName();
74          } catch (UnsupportedOperationException | IOException | SecurityException e) {
75              return NativeProcess.UNKNOWN_OWNER;
76          }
77      }
78  }
