1   package cx.corp.lacuna.core.linux;
2   
3   import java.io.File;
4   import java.io.FileFilter;
5   
6   class ProcFileFilter implements FileFilter {
7   
8       private final int pidMax;
9   
10      public ProcFileFilter(int pidMax) {
11          if (pidMax <= LinuxConstants.LOWEST_LEGAL_PID) {
12              throw new IllegalArgumentException("pidMax cannot be lower than or equal to 1!");
13          }
14          this.pidMax = pidMax;
15      }
16  
17      @Override
18      public boolean accept(File pathname) {
19          if (pathname == null) {
20              return false;
21          }
22  
23          return accept(pathname.isDirectory(), pathname.getName());
24      }
25  
26      boolean accept(boolean isDirectory, String fileName) {
27          if (!isDirectory) {
28              return false;
29          }
30  
31          Integer numericalPid = tryParseInt(fileName);
32          return numericalPid != null
33              && numericalPid >= LinuxConstants.LOWEST_LEGAL_PID
34              && numericalPid < pidMax;
35      }
36  
37      private Integer tryParseInt(String value) {
38          try {
39              return Integer.parseInt(value);
40          } catch (NumberFormatException ex) {
41              return null;
42          }
43      }
44  }
