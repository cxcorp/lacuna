1   package cx.corp.lacuna.core.windows;
2   
3   import java.util.Objects;
4   import java.util.Optional;
5   
6   public class WindowsProcessOwnerGetter implements ProcessOwnerGetter {
7   
8       private final ProcessTokenOpener tokenOpener;
9       private final TokenUserFinder tokenUserFinder;
10      private final TokenOwnerNameFinder tokenOwnerFinder;
11  
12      /**
13       * Constructs a new {@code WindowsProcessOwnerGetter} with the specified
14       * process token opener, token user finder, and user name finder.
15       * @param tokenOpener the process token opener.
16       * @param userFinder the token user finder.
17       * @param nameFinder the user name finder.
18       * @throws IllegalArgumentException if any of the arguments are null.
19       */
20      public WindowsProcessOwnerGetter(ProcessTokenOpener tokenOpener,
21                                       TokenUserFinder userFinder,
22                                       TokenOwnerNameFinder nameFinder) {
23          Objects.requireNonNull(tokenOpener, "tokenOpener cannot be null!");
24          Objects.requireNonNull(userFinder, "userFinder cannot be null!");
25          Objects.requireNonNull(nameFinder, "nameFinder cannot be null!");
26          this.tokenOpener = tokenOpener;
27          this.tokenUserFinder = userFinder;
28          this.tokenOwnerFinder = nameFinder;
29      }
30  
31      @Override
32      public Optional<String> get(ProcessHandle processHandle) {
33          Objects.requireNonNull(processHandle, "processHandle cannot be null!");
34  
35          try (ProcessToken token = tokenOpener.openToken(processHandle)) {
36              return tokenUserFinder.findTokenUser(token).flatMap(tokenOwnerFinder::getUserName);
37          } catch (TokenOpenException ex) {
38              // loggerino
39              return Optional.empty();
40          }
41      }
42  }
