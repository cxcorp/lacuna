1   package cx.corp.lacuna.core.linux;
2   
3   import cx.corp.lacuna.core.MemoryAccessException;
4   import cx.corp.lacuna.core.RawMemoryWriter;
5   import cx.corp.lacuna.core.domain.NativeProcess;
6   
7   import java.io.IOException;
8   import java.nio.ByteBuffer;
9   import java.nio.channels.SeekableByteChannel;
10  import java.util.Objects;
11  
12  /**
13   * {@inheritDoc}
14   *
15   * <p>The {@link LinuxRawMemoryWriter} class reads a process's memory using
16   * a {@link ReadableMemoryProvider}.
17   */
18  public class LinuxRawMemoryWriter implements RawMemoryWriter {
19  
20      private final WritableMemoryProvider memoryProvider;
21  
22      /**
23       * Constructs a new {@link LinuxRawMemoryWriter} with the specified
24       * readable memory provider.
25       * @param memoryProvider the memory provider.
26       * @throws NullPointerException if {@code memoryProvider} is null.
27       */
28      public LinuxRawMemoryWriter(WritableMemoryProvider memoryProvider) {
29          if (memoryProvider == null) {
30              throw new NullPointerException("Memory provider cannot be null!");
31          }
32          this.memoryProvider = memoryProvider;
33      }
34  
35      @Override
36      public void write(NativeProcess process, int offset, byte[] buffer) throws MemoryAccessException {
37          Objects.requireNonNull(process, "process cannot be null!");
38          Objects.requireNonNull(buffer, "buffer cannot be null!");
39          if (buffer.length < 1) {
40              throw new IllegalArgumentException("Cannot write fewer than 1 byte!");
41          }
42  
43          try (SeekableByteChannel output = memoryProvider.openWrite(process.getPid())) {
44  
45              long bytesToSkip = 0xFFFFFFFFL & offset; // interpret the offset as an unsigned value
46              output.position(bytesToSkip);
47  
48              int bytesWritten = output.write(ByteBuffer.wrap(buffer));
49              if (bytesWritten != buffer.length) {
50                  throw new MemoryAccessException("Only " + bytesWritten + " bytes out of " + buffer.length + " could be written!");
51              }
52          } catch (IOException ex) {
53              throw new MemoryAccessException("Writing process memory failed, see getCause()!", ex);
54          }
55      }
56  }
