1   package cx.corp.lacuna.core.windows;

3   import com.sun.jna.Native;
4   import com.sun.jna.ptr.IntByReference;
5   import cx.corp.lacuna.core.windows.winapi.Advapi32;
6   import cx.corp.lacuna.core.windows.winapi.SystemErrorCode;
7   import cx.corp.lacuna.core.windows.winapi.WinApiConstants;
8   
9   import java.util.Objects;
10  import java.util.Optional;
11  
12  /**
13   * Looks up the user name of a {@link Advapi32.TokenUser} via the Windows
14   * API {@link Advapi32} library.
15   * @cx.useswinapi
16   */
17  public class TokenOwnerNameFinder {
18  
19      private final Advapi32 advapi;
20  
21      /**
22       * Constructs a new {@code TokenOwnerNameFinder} with the specified
23       * Windows API {@code Advapi32} proxy.
24       * @param advapi the Windows API {@code Advapi32} proxy.
25       * @cx.useswinapi
26       */
27      public TokenOwnerNameFinder(Advapi32 advapi) {
28          Objects.requireNonNull(advapi, "advapi cannot be null!");
29          this.advapi = advapi;
30      }
31  
32      /**
33       * Gets the user name of the specified {@code TokenUser} account.
34       * <p>The method functions by calling the Windows API
35       * {@link Advapi32#lookupAccountSidW} method. The {@link Advapi32.TokenUser#user}
36       * field is passed as the SID.
37       * @param user the token user.
38       * @return the name of the user account, or {@link Optional#empty()} if the
39       *         lookup fails.
40       * @cx.useswinapi
41       */
42      public Optional<String> getUserName(Advapi32.TokenUser user) {
43          Objects.requireNonNull(user, "user cannot be null!");
44          // includes null terminator!
45          return getUsernameBufferLength(user)
46              .flatMap(bufLen -> lookupTokenUserName(user, bufLen));
47      }
48  
49      private Optional<Integer> getUsernameBufferLength(Advapi32.TokenUser user) {
50          IntByReference nameLength = new IntByReference(0);
51          IntByReference domainLength = new IntByReference(0);
52          IntByReference ignored = new IntByReference(0);
53  
54          boolean success =
55              advapi.lookupAccountSidW(
56                  WinApiConstants.NULL,
57                  user.user,
58                  null,
59                  nameLength,
60                  null,
61                  domainLength,
62                  ignored);
63  
64          return !success && callFailedBecauseBufferWasTooSmall()
65              ? Optional.of(nameLength.getValue())
66              : Optional.empty();
67      }
68  
69      private Optional<String> lookupTokenUserName(Advapi32.TokenUser user, int nameBufferLenWithNullTerminator) {
70          char[] nameBuffer = new char[nameBufferLenWithNullTerminator];
71          IntByReference nameBufNeededLen = new IntByReference(nameBuffer.length);
72          char[] domainBuffer = new char[WinApiConstants.MAX_DOMAIN_NAME_LENGTH];
73          IntByReference domainBufNeededLen = new IntByReference(domainBuffer.length);
74          IntByReference ignored = new IntByReference(0);
75  
76          boolean success =
77              advapi.lookupAccountSidW(
78                  WinApiConstants.NULL,
79                  user.user,
80                  nameBuffer,
81                  nameBufNeededLen,
82                  domainBuffer,
83                  domainBufNeededLen,
84                  ignored);
85  
86          int lengthWithoutNullTerminator = nameBufferLenWithNullTerminator - 1;
87          return success
88              ? Optional.of(new String(nameBuffer, 0, lengthWithoutNullTerminator))
89              : Optional.empty();
90      }
91  
92      private boolean callFailedBecauseBufferWasTooSmall() {
93          return Native.getLastError() == SystemErrorCode.INSUFFICIENT_BUFFER.getSystemErrorId();
94      }
95  }
