1   package cx.corp.lacuna.core.windows.winapi;
2   
3   import com.sun.jna.FunctionMapper;
4   import com.sun.jna.Library;
5   import com.sun.jna.Native;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   
10  /**
11   * Bootstraps Windows API proxy interfaces for use.
12   *
13   * <p>This class configures a function name mapper using {@link CamelToPascalCaseFunctionMapper}
14   * and lazily loads the native DLLs when called.
15   */
16  public class WinApiBootstrapper {
17  
18      private final Map<String, Object> loadLibraryOptions;
19  
20      public WinApiBootstrapper() {
21          loadLibraryOptions = new HashMap<>();
22          // Use a mapper so that we can use Java-style function names in the interfaces
23          FunctionMapper nameMapper = new CamelToPascalCaseFunctionMapper();
24          loadLibraryOptions.put(Library.OPTION_FUNCTION_MAPPER, nameMapper);
25      }
26  
27      public Kernel32 getKernel32() {
28          return Native.loadLibrary("Kernel32", Kernel32.class, loadLibraryOptions);
29      }
30  
31      public Advapi32 getAdvapi32() {
32          return Native.loadLibrary("Advapi32", Advapi32.class, loadLibraryOptions);
33      }
34  
35      public Psapi getPsapi() {
36          return Native.loadLibrary("Psapi", Psapi.class, loadLibraryOptions);
37      }
38  }
