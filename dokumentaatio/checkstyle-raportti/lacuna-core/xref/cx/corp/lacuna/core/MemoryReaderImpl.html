10  /**
11   * {@inheritDoc}
12   * @cx.littleendian
13   */
14  public class MemoryReaderImpl implements MemoryReader {
15  
16      private final RawMemoryReader rawMemoryReader;
17  
18      /**
19       * Constructs a new {@code MemoryReaderImpl} with the specified raw memory
20       * reader.
21       * @param rawMemoryReader the raw memory reader.
22       * @throws NullPointerException if {@code rawMemoryReader} is null.
23       */
24      public MemoryReaderImpl(RawMemoryReader rawMemoryReader) {
25          Objects.requireNonNull(rawMemoryReader, "rawMemoryReader cannot be null!");
26          this.rawMemoryReader = rawMemoryReader;
27      }
28  
29      @Override
30      public boolean readBoolean(NativeProcess process, int offset) {
31          byte readByte =
32              rawMemoryReader.read(process, offset, TypeSize.BOOLEAN.getSize()).get();
33          return readByte != 0;
34      }
35  
36      @Override
37      public byte readByte(NativeProcess process, int offset) {
38          return rawMemoryReader.read(process, offset, TypeSize.BYTE.getSize()).get();
39      }
40  
41      @Override
42      public char readCharUTF8(NativeProcess process, int offset) {
43          return (char) rawMemoryReader.read(process, offset, TypeSize.CHAR_UTF8.getSize()).get();
44      }
45  
46      @Override
47      public char readCharUTF16LE(NativeProcess process, int offset) {
48          return rawMemoryReader.read(process, offset, TypeSize.CHAR_UTF16LE.getSize()).getChar();
49      }
50  
51      @Override
52      public short readShort(NativeProcess process, int offset) {
53          return rawMemoryReader.read(process, offset, TypeSize.SHORT.getSize()).getShort();
54      }
55  
56      @Override
57      public int readInt(NativeProcess process, int offset) {
58          return rawMemoryReader.read(process, offset, TypeSize.INT.getSize()).getInt();
59      }
60  
61      @Override
62      public float readFloat(NativeProcess process, int offset) {
63          return rawMemoryReader.read(process, offset, TypeSize.FLOAT.getSize()).getFloat();
64      }
65  
66      @Override
67      public long readLong(NativeProcess process, int offset) {
68          return rawMemoryReader.read(process, offset, TypeSize.LONG.getSize()).getLong();
69      }

71      @Override
72      public double readDouble(NativeProcess process, int offset) {
73          return rawMemoryReader.read(process, offset, TypeSize.DOUBLE.getSize()).getDouble();
74      }
75  
76      @Override
77      public String readStringUTF8(NativeProcess process, int offset, int maxCodeUnitsToRead) {
78          if (maxCodeUnitsToRead < 1) {
79              throw new IllegalArgumentException("Cannot read strings shorter than 1 character!");
80          }
81  
82          byte[] buffer = new byte[maxCodeUnitsToRead];
83          int bytesRead = 0;
84  
85          for (int i = 0; i < maxCodeUnitsToRead; i++) {
86              byte readByte = readByte(process, offset + i);
87              if (readByte == 0) {
88                  // read until null character is met or maxLength is met
89                  break;
90              }
91              buffer[i] = readByte;
92              bytesRead++;
93          }
94  
95          return new String(buffer, 0, bytesRead, StandardCharsets.UTF_8);
96      }
97  
98      @Override
99      public String readStringUTF16LE(NativeProcess process, int offset, int maxCodeUnitsToRead) {
100         if (maxCodeUnitsToRead < 1) {
101             throw new IllegalArgumentException("Cannot read strings shorter than 1 character!");
102         }
103 
104         final int charSize = TypeSize.CHAR_UTF16LE.getSize();
105         int totalByteSize = maxCodeUnitsToRead * charSize;
106         ByteBuffer buffer = ByteBuffer.allocate(totalByteSize);
107         buffer.order(ByteOrder.LITTLE_ENDIAN);
108 
109         for (int i = 0; i < totalByteSize; i += charSize) {
110             short readShort = readShort(process, offset + i);
111             if (readShort == 0) {
112                 // read until null character is met or maxLength is met
113                 break;
114             }
115             buffer.putShort(readShort);
116         }
117 
118         buffer.flip();
119         byte[] truncatedBuf = new byte[buffer.remaining()];
120         buffer.get(truncatedBuf);
121         return new String(truncatedBuf, StandardCharsets.UTF_16LE);
122     }
123 
124     @Override
125     public byte[] read(NativeProcess process, int offset, int bytesToRead) {
126         if (bytesToRead < 1) {
127             throw new IllegalArgumentException("Number of bytes to read must be greater than zero");
128         }
129 
130         ByteBuffer buffer = rawMemoryReader.read(process, offset, bytesToRead);
131         byte[] ret = new byte[buffer.remaining()];
132         buffer.get(ret);
133         return ret;
134     }
135 }
