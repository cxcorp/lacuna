5   import java.nio.ByteBuffer;
6   
7   /**
8    * Provides functionality for reading a region of bytes from a native process's
9    * memory.
10   */
11  public interface RawMemoryReader {
12      /**
13       * Reads the specified amount of bytes from the specified offset at the native
14       * process' memory.
15       * @param process the native process whose memory will be read.
16       * @param offset the memory address offset to read from. This value is
17       *               interpreted as an unsigned value, meaning that negative
18       *               values are allowed.
19       * @param bytesToRead the amount of bytes to read.
20       * @return the read bytes.
21       * @throws MemoryAccessException if reading fails due to, for example, an
22       *                               access violation or insufficient rights, or
23       *                               if the read request was only partially completed.
24       * @throws NullPointerException if {@code process} is null.
25       * @throws IllegalArgumentException if {@code bytesToRead} is negative.
26       */
27      ByteBuffer read(NativeProcess process, int offset, int bytesToRead) throws MemoryAccessException;
28  }
